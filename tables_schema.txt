user_table:
    user_id(Should be auto-inc primary key)
    email_id(should be unique and not null)
    password_hash(password_hash)
    created_at (Timestamp)

user_profile_table:
    user_id(refer from user_table and acts as primary key)
    display_name(unique and not null)
    bio(text)

user_preferences_rating:
    user_id(refer from user_table and acts as primary key)
    15 columns each representing a hobby of the user

block_list:
    block_id(primary key)
    blocker_id(refers from user_id)
    blocked_id(can be a user_id or group_id)

friendship_table1:
    f_id:(primary key)
    sender_id(refers from user_id)
    receiver_id(refers from user_id)

friendship_table2:
    f_id:(primary key)
    status: 'pending','accepted', 'rejected', 'blocked'

identity_reveals_table1:
    reveal_id: (primary key)
    sender_id: (refers user_id)
    receiver_id: (refers to user_id)

identity_reveals_table2:
    reveal_id:(primary key refers from identity_reveals_table1)
    status: 'pending', 'accepted', 'rejected', 'blocked'

groups:
    group_id(primary key that auto-inc)
    group_name(unique and not null)
    description(text)

group_members:
    group_id(primary key refers from groups)
    members(array of members of a group)

messages:
    message_id(primary key that auto-inc)
    sender_id(refers from user_id)
    receiver_id(can be user_id or group_id)
    reciever_type(can be either 'user' or 'group')
    content(text not null)
    is_read(boolean)
    created_at(Timestamp)

user_recommendations:
    recommendation_id(primary key)
    user_id(refers from user_id)
    recommended_id(user_id or group_id)
    score(decimal score)
    created_at(Timestamp)
    status: 'pending', 'accepted', 'rejected', 'blocked'
